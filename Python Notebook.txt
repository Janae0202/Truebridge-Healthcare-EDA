df = pd.read_csv("prenatal_birth_data.csv")

# Remove records missing gestational age
df = df.dropna(subset=['Average_LMP_Gestational_Age_CLEANED'])

from scipy import stats
import numpy as np

# Correct column names
prenatal_care_code_col = 'Month Prenatal Care Began Code'
gestational_age_col = 'Average_LMP_Gestational_Age_CLEANED'

# Define conditions for 'early' and 'delayed' prenatal care using the code column
# Assuming codes 1, 2, 3 correspond to early care (1st, 2nd, 3rd month).
# Other codes (0 and 4-9) correspond to delayed or no care.
conditions = [
    df[prenatal_care_code_col].isin([1.0, 2.0, 3.0]),
    df[prenatal_care_code_col].isin([0.0, 4.0, 5.0, 6.0, 7.0, 8.0, 9.0])
]
choices = ['early', 'delayed']

# Create the new 'prenatal_care_category' column
# Default to None for codes not covered (e.g., if there are other codes or NaNs in the code column)
df['prenatal_care_category'] = np.select(conditions, choices, default=None)

# Filter out NaNs in the category or gestational age
df_filtered = df.dropna(subset=['prenatal_care_category', gestational_age_col])

# Prepare data for t-test
early_group = df_filtered[df_filtered['prenatal_care_category'] == 'early'][gestational_age_col]
delayed_group = df_filtered[df_filtered['prenatal_care_category'] == 'delayed'][gestational_age_col]

# Initialize t_stat and p_value in case groups are empty
t_stat = np.nan
p_value = np.nan

# Perform t-test only if both groups have enough data
if len(early_group) > 1 and len(delayed_group) > 1:
    # Welch’s t-test (robust to unequal variances)
    t_stat, p_value = stats.ttest_ind(early_group, delayed_group, equal_var=False)
    print(f"T-statistic: {t_stat:.2f}, P-value: {p_value:.3f}")
else:
    print("Not enough data in one or both groups ('early' or 'delayed') to perform t-test. Please check your categorization or data.")

t_stat, p_value

# Group gestational age by education
groups = [
    df[df["Mother's Education"] == level]['Average_LMP_Gestational_Age_CLEANED']
    for level in df["Mother's Education"].dropna().unique()
]

stats.f_oneway(*groups)

for r in df['Census Region'].unique():
    subset = df[df['Census Region'] == r]
    early = subset[subset['prenatal_care_category'] == 'early']['Average_LMP_Gestational_Age_CLEANED']
    delayed = subset[subset['prenatal_care_category'] == 'delayed']['Average_LMP_Gestational_Age_CLEANED']

    # Perform t-test only if both groups have enough data for the current region
    if len(early) > 1 and len(delayed) > 1:
        t_stat, p = stats.ttest_ind(early, delayed, equal_var=False)
        print(f"Region: {r} | p-value: {p:.4f}")
    else:
        print(f"Region: {r} | Not enough data in one or both groups ('early' or 'delayed') to perform t-test.")
